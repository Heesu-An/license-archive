## **2\. Filtering & Sorting**

```
#### 인덱싱 ####
df['열이름']         # 특정 열 추출
df.열이름            # 특정 열 추출
df.iloc[1]          # 1행 인덱싱
df.iloc[0:2, 0:2]   # 0~1행, 0~1열 인덱싱
df[df["age"] > 20]  # 조건을 만족하는 행 인덱싱

#### 조건 데이터 #### 🚨🚨🚨🚨
df[df['item_name'].str.contains("Chips")]    # item_name 열에서 Chips를 포함하는 경우
df['item_price'].str[1:].astype('float')
df[df['new_price'].isin(리스트)]               # new_price 컬럼에서 리스트 내에 있는 값이라면 추출

#### 컬럼 내 중복 제거 🚨🚨####
df.drop_duplicates("열이름", keep="last" 혹은 "first")  # 열 내 중복 제거. 여러 개 중 첫번째 혹은 마지막 꺼 선택

#### 정렬 ####
df.sort_values("기준열", ascending=True)   # 컬럼 기준 오름차순 정렬
df.sort_index(ascending=True)             # 인덱스 기준 오름차순 정렬

#### 결측치 ####
print(df.isnull())                       # 결측치 여부 True, False
print(df.isnull().sum())                 # 각 컬럼의 결측치 개수
df.loc[df['choice_description'].isnull(), 'choice_description'] = "NoData"  # 🚨🚨🚨🚨 특정 행의 결측값 대체
df.dropna(subset=['열1', '열2'])          # 지정한 열 결측치 삭제 / 지정 안하면 결측치 있는 행 삭제
df.fillna(method="ffill").fillna(method="bfill")     # 직전 값으로 대체. 첫번째 행은 뒤의 값으로 대체
```

## **3. Grouping**

```
#### 그룹핑 #### 🚨🚨🚨🚨
df.groupby("team")["score"].mean()  # 'team'컬럼을 기준으로 같은 그룹끼리 묶고, score 열을 평균냄.
df.groupby("team")["score"].mean().rest_index()   # 'team'컬럼을 인덱스가 아닌 열로 바꿈. (우리가 흔히 아는 데이터프레임과 숫자 인덱스로 변환)
df.groupby("team").filter(lambda x: x["score"].mean() >= 70)  # 그룹의 평균 score가 70 이상인 팀만
df.groupby(["team", "city"])["score"].mean()  # 'team'컬럼 기준으로 나눈 그룹 안에서, city 컬럼 기준으로 또 나눔

#### 컬럼 내 그룹 별 개수 ####
df['AgeState'].value_counts()    # 컬럼 내 카테고리 별 개수
df.groupby("host_name").size().sort_values()   # 🚨🚨🚨🚨 size() 함수: 그룹 내 카테고리 별 개수
df2 = pd.DataFrame(df.groupby("host_name").size().sort_values(ascending=False), columns=['counts'])
pd.DataFrame(df.groupby("host_name").size().sort_values(ascending=False)).rename(columns={0:'counts'})
df.groupby(["neighbourhood_group", "neighbourhood"], as_index=False).size()  # as_index 매개변수!!
df.groupby(['neighbourhood_group', 'neighbourhood'], as_index=False).size().groupby("neighbourhood_group", as_index=False).max()   # 🚨🚨🚨🚨 "neighbourhood_group의 값에 따른 neighbourhood컬럼 값 중 neighbourhood_group그룹의 최댓값들을 출력하라"
df[['neighbourhood_group', 'price']].groupby("neighbourhood_group", as_index=False).agg(['mean', 'var', 'max', 'min'])    # 🚨🚨🚨🚨 "neighbourhood_group 값에 따른 price값의 평균, 분산, 최대, 최소 값을 구하여라"

tmp = df[['neighbourhood_group', 'room_type']].groupby(['neighbourhood_group', 'room_type']).size().unstack()  # 🚨🚨🚨🚨 "55. 데이터중 neighbourhood_group 값에 따른 room_type 컬럼의 숫자를 구하고 neighbourhood_group 값을 기준으로 각 값의 비율을 구하여라"
tmp.loc[:, :] = tmp.values / tmp.values.sum(axis=1).reshape(-1, 1)
print(tmp)


##### 🚨🚨🚨🚨 apply 함수는 기본적으로 axis=0임. df['name'].apply처럼 열 하나에 apply 함수를 적용하는 것은 시리즈에 적용하는 것이므로 우리가 볼 때는 행 단위로 적용하는 것처럼 보임. 그러나 df[['name', 'age']].apply처럼 적용할 때는 열 단위로 적용되므로 행 단위로 진행하고 싶으면 axis 반드시 지정하기
```

df.Yr_Mo_Dy.dt.year   (month, day, hour, minute, second)      # 시리즈에서 연월일 추출 (year, month, day)
x.year                                                        # x가 datetime64 타입일 때, 연월일 추출
df.Yr_Mo_Dy.dt.dayofweek                                      # 요일 숫자로 반환

df.resample("W").agg(['min','max','mean','std'])              # 인덱스가 datetime 형식일 때, 주 단위로 리샘플링하여 계산 ("A", "AS", "M", "MS", "W", "D", "Q", "QS")

## **5\. Merge , Concat**

```
- merge: 인덱스, 공통 컬럼을 기준으로 합침
- concat: 갖다 붙임 ---> 추천

#### concat 파라미터 ####
axis = 0(행), 1(열)
join = 'outer'(기본값, 합집합), 'inner'(교집합)

#### concat ####
pd.concat([df1, df2], join='inner')
pd.concat([df1, df2], join='inner', axis=1)

#### merge ####
pd.merge(df1, df2, on='공통열', how='inner')
pd.merge(df1, df2, left_on='df1열', right_on="df2열" how='outer')
```
